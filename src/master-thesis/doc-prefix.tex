%%\newcounter{example}[section]\setcounter{example}{0}
\providecommand{\theexample}{}
\renewcommand{\theexample}{\arabic{section}.\arabic{example}}
\renewenvironment{myexample}[2][]
  {\refstepcounter{example}
  \mdfsetup{frametitlefont=\normalfont,
    topline=false,
    bottomline=false,
    rightline=false,
    linecolor=yellow!10!orange!20,
    linewidth=2pt,
    skipabove=12,
    %needspace=4\baselineskip,
    innertopmargin=1.2\baselineskip,
    singleextra={
      \node[
        overlay,
        outer sep=0pt,
        right,
        xshift=-1.pt,
        %anchor=north east,
        %text width=2.5cm,
        minimum height=4ex,
        fill=yellow!10!orange!20,
      ] at (O|-P)%
      {\strut \textbf{Example~\theexample}};
      },
    firstextra={
      \node[
        overlay,
        outer sep=0pt,
        right,
        xshift=-1pt,
        %anchor=north east,
        %text width=2.5cm,
        minimum height=4ex,
        fill=yellow!10!orange!20,
      ] at (O|-P)%
      {\strut \textbf{Example~\theexample}};
      }}
  \vspace{0.2cm}\begin{mdframed}
  }
  {\end{mdframed}}

\newcounter{axiom}[section]\setcounter{axiom}{0}
\renewcommand{\theaxiom}{\arabic{section}.\arabic{axiom}}
\newenvironment{axiom}[2][]{%
  \refstepcounter{axiom}%
  \mdfsetup{%
    singleextra={%
      \node[
        overlay,
        xshift=10pt,
        right,
        outer sep=0pt,
        rectangle,
        fill=green!20
      ] at (O|-P)
      {\strut \textbf{Axiom~\theaxiom\ifstrempty{#1}{}{:~#1}}
      };
    },
    firstextra={%
      \node[
        overlay,
        xshift=10pt,
        right,
        outer sep=0pt,
        rectangle,
        fill=green!20
      ] at (O|-P)
      {\strut \textbf{Axiom~\theaxiom\ifstrempty{#1}{}{:~#1}}
      };
    }
  }
  \mdfsetup{
    innertopmargin=1.2\baselineskip,linecolor=green!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
  }
  \begin{mdframed}[skipabove=20pt]\relax%
  \label{#2}
}
{\end{mdframed}}
